  1. In Anlehnung an den Anwendungsfall im dritten Notebook, lasse Dir sämtliche Viertelstunden eines kompletten Tages ausgeben, also:

    00:00
    00:15

min_per_hour = 59
hours_per_day = 24

hour = 0
while hour < hours_per_day:
    
    if len(str(hour)) == 1:
        hour_str = "0" + str(hour)
    else:
        hour_str = str(hour) 

    minute = 0
    
    while minute < min_per_hour:
        
        if len(str(minute)) == 1:
            print(hour_str + ":0" + str(minute))
        else:
            print(hour_str + ":" + str(minute))
        
        minute += 15
    
    hour+=1
